Portal Setup WiFi ESP8266

Jika ESP belum punya konfigurasi WiFi, modul ini membuat hotspot SmartHydro-Setup dan halaman setup di 192.168.4.1.

Simpan Config

SSID dan password disimpan ke wifi_config.json, sehingga coding utama tidak perlu diubah saat pindah jaringan.

Mode AP

Jika koneksi gagal, ESP membuat access point untuk menerima input WiFi baru dari browser.

Reset WiFi

Command RESET_WIFI dari MQTT dapat memanggil delete_wifi_config(), lalu ESP restart dan masuk mode setup lagi.

import network
import time
import ujson
import socket
import os
import gc
import machine
from machine import Pin

CONFIG_FILE = "wifi_config.json"
AP_SSID = "SmartHydro-Setup"
AP_PASSWORD = "12345678"
WIFI_TIMEOUT = 20

def url_decode(text):
    text = text.replace("+", " ")
    result = ""
    i = 0

    while i < len(text):
        if text[i] == "%" and i + 2 < len(text):
            try:
                result += chr(int(text[i + 1:i + 3], 16))
                i += 3
            except ValueError:
                result += text[i]
                i += 1
        else:
            result += text[i]
            i += 1

    return result

def get_param(query, name):
    items = query.split("&")
    for item in items:
        if "=" in item:
            key, value = item.split("=", 1)
            if key == name:
                return url_decode(value)
    return ""

def load_wifi_config():
    try:
        with open(CONFIG_FILE, "r") as file:
            data = ujson.load(file)
        ssid = data.get("ssid", "")
        password = data.get("password", "")
        if ssid != "":
            return ssid, password
    except OSError:
        pass
    return None, None

def save_wifi_config(ssid, password):
    with open(CONFIG_FILE, "w") as file:
        ujson.dump({"ssid": ssid, "password": password}, file)

def delete_wifi_config():
    try:
        os.remove(CONFIG_FILE)
        print("WiFi config dihapus")
    except OSError:
        print("WiFi config belum ada")

def start_wifi_manager(led_mqtt):
    print("\n====================\n WIFI MANAGER AKTIF\n====================")
    led_mqtt.value(0)
    gc.collect()

    ap = network.WLAN(network.AP_IF)
    ap.active(True)
    try:
        ap.config(essid=AP_SSID, password=AP_PASSWORD, authmode=network.AUTH_WPA_WPA2_PSK)
    except Exception:
        ap.config(essid=AP_SSID) # Fallback jika modul tidak mendukung enkripsi tertentu

    server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
    server.bind(("", 80))
    server.listen(1)

    while True:
        conn = None
        try:
            gc.collect()
            conn, addr = server.accept()
            request = conn.recv(1024).decode()

            if "GET /save?" in request:
                path = request.split(" ")[1]
                query = path.split("?", 1)[1]
                ssid_baru = get_param(query, "ssid")
                password_baru = get_param(query, "password")

                if ssid_baru != "":
                    save_wifi_config(ssid_baru, password_baru)
                    conn.send(halaman_wifi_manager("WiFi tersimpan. ESP restart...").encode())
                    conn.close()
                    time.sleep(2)
                    machine.reset()
                else:
                    conn.send(halaman_wifi_manager("SSID tidak boleh kosong!").encode())
            else:
                conn.send(halaman_wifi_manager().encode())

            conn.close()
        except Exception as e:
            if conn:
                conn.close()

def connect_wifi(led_mqtt):
    wlan = network.WLAN(network.STA_IF)
    wlan.active(True)
    ssid, password = load_wifi_config()

    if ssid is None:
        print("Config WiFi tidak ditemukan. Menjalankan Mode AP.")
        start_wifi_manager(led_mqtt)

    print("Menghubungkan ke WiFi:", ssid)
    wlan.connect(ssid, password)
    timeout = WIFI_TIMEOUT

    while not wlan.isconnected() and timeout > 0:
        time.sleep(1)
        timeout -= 1

    if wlan.isconnected():
        print("WiFi terhubung. IP:", wlan.ifconfig()[0])
        try:
            network.WLAN(network.AP_IF).active(False)
        except Exception:
            pass
        return wlan
    else:
        print("Gagal terhubung ke WiFi. Menjalankan Mode AP.")
        start_wifi_manager(led_mqtt)

Penjelasan Detail Coding

import network

Library utama untuk mengatur WiFi pada ESP8266. Mode STA_IF dipakai untuk konek ke router, sedangkan AP_IF dipakai untuk membuat hotspot setup.

CONFIG_FILE = "wifi_config.json"

Nama file tempat ESP menyimpan SSID dan password. Setelah file ini ada, ESP bisa langsung konek WiFi tanpa mengisi form lagi.

AP_SSID dan AP_PASSWORD

Ini adalah nama dan password hotspot sementara dari ESP. Pengguna connect ke hotspot ini, lalu membuka 192.168.4.1 untuk mengisi WiFi asli.

url_decode(text)

Form HTML mengirim spasi sebagai + dan karakter khusus sebagai kode persen seperti %20. Fungsi ini mengubah format URL tersebut kembali menjadi teks normal.

load_wifi_config()

Fungsi ini membaca wifi_config.json dengan ujson.load(), lalu mengambil ssid dan password. Jika SSID ada, fungsi mengembalikan pasangan ssid, password. Jika file belum ada atau SSID kosong, fungsi mengembalikan None, None.

save_wifi_config(ssid, password)

SSID dan password langsung disimpan ke wifi_config.json memakai ujson.dump(). Setelah tersimpan, ESP bisa memakai data ini lagi saat boot berikutnya.

delete_wifi_config()

Fungsi ini menghapus file konfigurasi WiFi. Dipakai saat pengguna mengirim command RESET_WIFI dari Telegram/MQTT.

halaman_wifi_manager(pesan="")

Fungsi ini menghasilkan halaman HTML sederhana berisi form SSID dan password. Teks pesan dipakai untuk menampilkan status seperti "WiFi tersimpan" atau "SSID tidak boleh kosong".

get_param(query, name)

Fungsi ini membaca query string dari form, memecahnya berdasarkan tanda &, lalu mencari parameter sesuai nama yang diminta, misalnya ssid atau password.

start_wifi_manager(led_mqtt)

Fungsi ini mengaktifkan mode access point, mematikan LED MQTT, menjalankan garbage collection, membuat web server di port 80, menerima request browser, menyimpan data WiFi, lalu restart ESP setelah konfigurasi berhasil.

server.accept()

Baris ini menunggu koneksi dari browser. Setelah browser mengirim request, ESP membaca request tersebut dan menentukan apakah pengguna membuka halaman utama atau menekan tombol simpan.

connect_wifi(led_mqtt)

Fungsi utama yang dipanggil oleh main.py. Fungsi ini membaca config, masuk WiFi Manager jika SSID belum ada, mencoba konek ke router, mematikan mode AP jika berhasil, lalu mengembalikan object wlan.

WIFI_TIMEOUT

Nilai ini menjadi hitung mundur koneksi WiFi. Jika ESP belum terkoneksi saat timeout habis, program menjalankan start_wifi_manager(led_mqtt) agar pengguna bisa memasukkan SSID dan password baru.
