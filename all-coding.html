main.py - ESP8266 Smart Hydroponic

import time
import socket
import gc
import machine
from machine import Pin
import dht
from umqtt.simple import MQTTClient
import wifimgr

MQTT_SERVER = "broker.emqx.io"
MQTT_PORT = 1883
MQTT_TOPIC = b"hanif/smarthydro"
MQTT_CONTROL_TOPIC = b"hanif/smarthydro/control"
CLIENT_ID = b"ESP8266-SmartHydro"

WRITE_API_KEY = "SHKXBHKDZLDLFT2X"
THINGSPEAK_HOST = "api.thingspeak.com"

DHT_PIN = Pin(14)
RELAY_PIN = Pin(5, Pin.OUT)

TRIG_PIN = Pin(16, Pin.OUT)
ECHO_PIN = Pin(4, Pin.IN)

LED_MQTT = Pin(12, Pin.OUT)
LED_POMPA_OFF = Pin(13, Pin.OUT)
LED_POMPA_ON = Pin(15, Pin.OUT)

TINGGI_WADAH_CM = 20
BATAS_JARAK_AIR_RENDAH = 15
BATAS_JARAK_AIR_CUKUP = 8

RELAY_ON = 0
RELAY_OFF = 1

suhu = 0
kelembaban = 0
jarak_air_cm = 0
level_air_persen = 0

status_pompa = "OFF"
mode_pompa = "AUTO"

INTERVAL_SEND = 20
INTERVAL_SENSOR = 5

dht_sensor = dht.DHT11(DHT_PIN)
mqtt_client = None

def setup_pin():
    RELAY_PIN.value(RELAY_OFF)
    TRIG_PIN.value(0)
    LED_MQTT.value(0)
    LED_POMPA_OFF.value(1)
    LED_POMPA_ON.value(0)

def update_led_pompa():
    if status_pompa == "ON":
        LED_POMPA_ON.value(1)
        LED_POMPA_OFF.value(0)
    else:
        LED_POMPA_ON.value(0)
        LED_POMPA_OFF.value(1)

def baca_jarak_once():
    try:
        TRIG_PIN.value(0)
        time.sleep_us(2)

        TRIG_PIN.value(1)
        time.sleep_us(10)
        TRIG_PIN.value(0)

        durasi = machine.time_pulse_us(ECHO_PIN, 1, 30000)

        if durasi < 0:
            return None

        jarak = durasi / 58
        return jarak

    except Exception as e:
        print("Error ultrasonic:", e)
        return None

def baca_ultrasonic():
    total = 0
    jumlah = 0

    for i in range(3):
        jarak = baca_jarak_once()

        if jarak is not None and 1 < jarak < 400:
            total += jarak
            jumlah += 1

        time.sleep_ms(70)

    if jumlah == 0:
        return None

    return total / jumlah

def hitung_level_air(jarak):
    if jarak is None:
        return 0

    tinggi_air = TINGGI_WADAH_CM - jarak

    if tinggi_air < 0:
        tinggi_air = 0
    elif tinggi_air > TINGGI_WADAH_CM:
        tinggi_air = TINGGI_WADAH_CM

    persen = int((tinggi_air * 100) / TINGGI_WADAH_CM)
    return persen

def mqtt_callback(topic, msg):
    global status_pompa, mode_pompa

    try:
        pesan = msg.decode().strip().upper()
    except Exception:
        print("Pesan MQTT tidak bisa dibaca")
        return

    print("\nPesan MQTT masuk")
    print("Topic:", topic)
    print("Pesan:", pesan)

    if topic != MQTT_CONTROL_TOPIC:
        return

    if "RESET_WIFI" in pesan:
        print("Reset WiFi...")
        wifimgr.delete_wifi_config()
        time.sleep(1)
        machine.reset()

    elif "AUTO" in pesan:
        mode_pompa = "AUTO"
        print("Mode pompa AUTO")

    elif "OFF" in pesan or "MATI" in pesan:
        mode_pompa = "MANUAL"
        status_pompa = "OFF"
        RELAY_PIN.value(RELAY_OFF)
        update_led_pompa()
        print("Pompa dimatikan manual")
        print("Relay value:", RELAY_PIN.value())

    elif "ON" in pesan or "NYALA" in pesan:
        mode_pompa = "MANUAL"
        status_pompa = "ON"
        RELAY_PIN.value(RELAY_ON)
        update_led_pompa()
        print("Pompa dinyalakan manual")
        print("Relay value:", RELAY_PIN.value())

    else:
        print("Perintah tidak dikenal")

def connect_mqtt():
    global mqtt_client

    try:
        print("\nMenghubungkan ke MQTT...")
        mqtt_client = MQTTClient(
            CLIENT_ID,
            MQTT_SERVER,
            port=MQTT_PORT,
            keepalive=60
        )

        mqtt_client.set_callback(mqtt_callback)
        mqtt_client.connect()
        mqtt_client.subscribe(MQTT_CONTROL_TOPIC)

        LED_MQTT.value(1)

        print("MQTT terhubung")
        print("Publish topic :", MQTT_TOPIC)
        print("Control topic :", MQTT_CONTROL_TOPIC)

        return mqtt_client

    except Exception as e:
        print("MQTT gagal:", e)
        LED_MQTT.value(0)
        return None

def baca_sensor():
    global suhu, kelembaban, jarak_air_cm, level_air_persen

    try:
        dht_sensor.measure()
        suhu = dht_sensor.temperature()
        kelembaban = dht_sensor.humidity()
    except Exception as e:
        print("Gagal baca DHT:", e)

    jarak = baca_ultrasonic()

    if jarak is not None:
        jarak_air_cm = round(jarak, 1)
        level_air_persen = hitung_level_air(jarak_air_cm)
    else:
        print("Gagal baca ultrasonic")
        jarak_air_cm = 0
        level_air_persen = 0

    print("\nData sensor")
    print("Suhu            :", suhu, "C")
    print("Kelembaban      :", kelembaban, "%")
    print("Jarak air       :", jarak_air_cm, "cm")
    print("Level air       :", level_air_persen, "%")
    print("Mode pompa      :", mode_pompa)
    print("Status pompa    :", status_pompa)
    print("Relay value     :", RELAY_PIN.value())

def kontrol_pompa_otomatis():
    global status_pompa

    if mode_pompa == "AUTO":
        if jarak_air_cm == 0:
            print("Data ultrasonic tidak valid, pompa tidak diubah")
            return

        if jarak_air_cm > BATAS_JARAK_AIR_RENDAH:
            RELAY_PIN.value(RELAY_ON)
            status_pompa = "ON"
            update_led_pompa()
            print("Auto: air rendah, pompa ON")

        elif jarak_air_cm < BATAS_JARAK_AIR_CUKUP:
            RELAY_PIN.value(RELAY_OFF)
            status_pompa = "OFF"
            update_led_pompa()
            print("Auto: air cukup, pompa OFF")

        else:
            print("Auto: level air sedang, status tetap")

def kirim_mqtt():
    global mqtt_client

    if mqtt_client is None:
        mqtt_client = connect_mqtt()

    if mqtt_client is None:
        print("MQTT belum terhubung")
        return

    payload = (
        f'{{"temperature":{suhu},"humidity":{kelembaban},"distance_cm":{jarak_air_cm},'
        f'"water_level_percent":{level_air_persen},"pump_status":"{status_pompa}","pump_mode":"{mode_pompa}"}}'
    )

    try:
        mqtt_client.publish(MQTT_TOPIC, payload)
        LED_MQTT.value(1)

        print("\nMQTT terkirim:")
        print(payload)

    except Exception as e:
        print("MQTT publish gagal:", e)
        LED_MQTT.value(0)

        try:
            mqtt_client.disconnect()
        except Exception:
            pass

        mqtt_client = None

def kirim_thingspeak():
    pompa_value = 1 if status_pompa == "ON" else 0
    mode_value = 1 if mode_pompa == "AUTO" else 0

    path = f"/update?api_key={WRITE_API_KEY}&field1={suhu}&field2={kelembaban}&field3={level_air_persen}&field4={pompa_value}&field5={mode_value}"

    s = None

    try:
        print("\nMengirim ke ThingSpeak...")
        addr = socket.getaddrinfo(THINGSPEAK_HOST, 80)[0][-1]
        s = socket.socket()
        s.settimeout(5)
        s.connect(addr)

        request = f"GET {path} HTTP/1.1\r\nHost: {THINGSPEAK_HOST}\r\nConnection: close\r\n\r\n"
        s.send(request.encode())
        s.recv(128)

        print("ThingSpeak terkirim")

    except Exception as e:
        print("ThingSpeak gagal:", e)

    finally:
        if s:
            try:
                s.close()
            except Exception:
                pass

setup_pin()

print("\nMemulai Smart Hydroponic")
print("DHT11    : D5 / GPIO14")
print("Relay    : D1 / GPIO5")
print("TRIG     : D0 / GPIO16")
print("ECHO     : D2 / GPIO4")
print("LED MQTT : D6 / GPIO12")
print("LED OFF  : D7 / GPIO13")
print("LED ON   : D8 / GPIO15")

wlan = wifimgr.connect_wifi(LED_MQTT)
mqtt_client = connect_mqtt()

last_send = time.time()
last_sensor = 0

print("\nSmart Hydroponic siap")

while True:
    gc.collect()

    if not wlan.isconnected():
        print("\nWiFi terputus, mencoba konek ulang...")
        LED_MQTT.value(0)

        try:
            if mqtt_client:
                mqtt_client.disconnect()
        except Exception:
            pass

        mqtt_client = None
        wlan = wifimgr.connect_wifi(LED_MQTT)

    if mqtt_client is not None:
        try:
            mqtt_client.check_msg()
        except Exception as e:
            print("Gagal cek MQTT:", e)
            LED_MQTT.value(0)

            try:
                mqtt_client.disconnect()
            except Exception:
                pass

            mqtt_client = None

    sekarang = time.time()

    if sekarang - last_sensor >= INTERVAL_SENSOR:
        baca_sensor()
        kontrol_pompa_otomatis()
        last_sensor = sekarang

    if sekarang - last_send >= INTERVAL_SEND:
        kirim_mqtt()
        kirim_thingspeak()
        last_send = sekarang

    time.sleep_ms(200)

  }

Penjelasan Alur Semua Coding

1. Import Library

time, dht, machine, socket, gc, MQTTClient, dan wifimgr dipakai untuk delay, sensor, pin ESP, HTTP ThingSpeak, manajemen memori, MQTT, dan koneksi WiFi.

2. Konfigurasi MQTT

Bagian MQTT_SERVER, MQTT_TOPIC, dan MQTT_CONTROL_TOPIC menentukan broker, topic pengiriman data, dan topic penerima command.

3. Konfigurasi ThingSpeak

WRITE_API_KEY menentukan channel ThingSpeak tujuan, sedangkan THINGSPEAK_HOST adalah server API yang menerima data.

4. Deklarasi Pin

DHT11, relay, ultrasonic, dan LED indikator disambungkan ke GPIO ESP8266. Output dipakai untuk relay, trigger, dan LED. Input dipakai untuk echo ultrasonic.

5. Variabel Status

Variabel seperti suhu, kelembaban, level_air_persen, status_pompa, dan mode_pompa menyimpan kondisi terakhir sistem agar bisa dikirim ke MQTT dan ThingSpeak. INTERVAL_SEND mengatur jeda kirim data, sedangkan INTERVAL_SENSOR mengatur jeda baca sensor.

6. Sensor Ultrasonic

baca_jarak_once() membaca satu kali jarak, baca_ultrasonic() mengambil rata-rata beberapa pembacaan, lalu hitung_level_air() mengubah jarak menjadi persen level air.

7. MQTT Callback

mqtt_callback() adalah pusat kontrol manual. Perintah dari Telegram lewat n8n masuk ke MQTT, lalu ESP mengubah mode pompa atau mereset WiFi sesuai isi pesan.

8. Kontrol Pompa Otomatis

kontrol_pompa_otomatis() hanya aktif saat mode AUTO. Air rendah membuat pompa ON, air cukup membuat pompa OFF, dan air sedang mempertahankan kondisi sebelumnya.

9. Kirim Data

kirim_mqtt() membuat payload JSON memakai f-string lalu mengirimnya ke broker. kirim_thingspeak() membuat URL update memakai f-string dan mengirim data ke dashboard grafik memakai HTTP GET.

10. Loop Utama

Program selalu mengecek koneksi WiFi, mengecek pesan MQTT, membaca sensor setiap INTERVAL_SENSOR, mengontrol pompa, mengirim data setiap INTERVAL_SEND, lalu mengulang dengan jeda pendek time.sleep_ms(200) agar respons MQTT tetap cepat.

11. n8n Monitoring

Function node pertama membaca JSON dari MQTT, membuat teks monitoring, dan menambahkan alert berdasarkan jarak HC-SR04: jarak besar berarti air rendah, jarak kecil berarti air cukup atau penuh.

12. n8n Command

Function node kedua membaca teks Telegram, menormalisasi command, lalu membuat output topic dan message untuk dikirim ke MQTT Publish node.
